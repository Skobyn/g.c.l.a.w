P — Pseudocode
  class LiveSpanProcessor(SpanProcessor):
      def on_start(self, span, parent_context):
          queue_for(run_id).put_nowait({"event": "span.start", ...})
      def on_end(self, span):
          queue_for(run_id).put_nowait({"event": "span.end", ...})
          if is_agent_span(span):
              upsert_firestore_agent_run(...)   # throttled

  FastAPI route:
    @router.get("/api/runs/{run_id}/events")
    async def events(run_id, request):
        assert request.state.user_id == run_owner(run_id)
        async def gen():
            async for evt in drain_queue(run_id):
                yield "event: " + evt.event + "\ndata: " + json.dumps(evt.data) + "\n\n"
        return StreamingResponse(gen(), media_type="text/event-stream")

P — Pseudocode
  # Inline guardrails
  class GuardrailService:
      def __init__(self, authority_profile):
          self.validators = build_validators_for(authority_profile)
      async def check_output(self, text, context) -> GuardrailResult:
          results = await asyncio.gather(*[v.check(text, context) for v in self.validators])
          return merge(results)

  # In AgentRunner, after event drain:
  result = await guardrail_service.check_output(final_text, retrieved_context)
  turn_span.set_attributes({
      "guardrail.outcome": result.outcome,    # pass | warn | block
      "guardrail.violations": json.dumps(result.violations),
  })
  if result.outcome == "block": raise GuardrailBlockedError(result)

  # Nightly async scoring
  # Cloud Scheduler -> /admin/eval/run
  async def run_nightly_scoring():
      spans = sample_recent_spans(n=100)
      scorer = GenAIEvaluationClient(project=settings.gcp_project)
      for span in spans:
          metrics = await scorer.score(
              responses=[span.attributes["output.value"]],
              contexts=[span.attributes.get("retrieval.context")],
              metrics=["groundedness", "coherence", "tool_use_quality"],
          )
          firestore.write("eval_results/" + span.span_id, metrics)
